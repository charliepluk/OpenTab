CREATE TABLE opentab.restaurants(
    restID INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    restName varchar(50) NOT NULL,
    restPhone varchar(14) NOT NULL UNIQUE,
    restEmail varchar(50) NOT NULL UNIQUE,
    restLocation varchar(100) NOT NULL UNIQUE,
    restDescription varchar(200),
    restOpenTime time(0),
    restCloseTime time(0)
);

=============
ORDERS TABLE
=============
INSERT INTO `orders`(`restID`, `customerID`, `orderNotes`) VALUES (1,1,'testnote')

CREATE TABLE opentab.orders(
    orderID INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    restID int,
	FOREIGN KEY (restID) REFERENCES openTab.restaurants(restID) ON DELETE CASCADE,
    customerID int,
	FOREIGN KEY (customerID) REFERENCES openTab.customers(customerID) ON DELETE CASCADE,
    orderNotes varchar(150),
    orderDateTime datetime DEFAULT CURRENT_TIMESTAMP
);

ITEMS TABLE
===========

CREATE TABLE opentab.items(
    itemID INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    restID int,
	FOREIGN KEY (restID) REFERENCES openTab.restaurants(restID) ON DELETE CASCADE,
    itemName varchar(25) NOT NULL,
    itemPrice decimal(6,2) NOT NULL
);

ORDERITEMS TABLE
=================
INSERT INTO `order_items`(`orderID`, `itemID`, `itemQuantity`, `totalPriceOfItems`) VALUES (4,1,2,25.38)

CREATE TABLE opentab.order_items(
    orderItemsID INT(10) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    orderID int,
	FOREIGN KEY (orderID) REFERENCES openTab.orders(orderID) ON DELETE CASCADE,
    itemID int,
	FOREIGN KEY (itemID) REFERENCES openTab.items(itemID) ON DELETE CASCADE,
    itemQuantity int(2) NOT NULL,
    totalPriceOfItems decimal(6,2) NOT NULL
);
